Enum gstd::errors::ExtError

pub enum ExtError {
    Decode,
    SyscallUsage,
    Memory(MemoryError),
    Message(MessageError),
    Wait(WaitError),
    Execution(ExecutionError),
}
Expand description

An error occurred in API.
