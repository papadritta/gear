Same as reply_commit, but with explicit gas limit.

Examples

use gcore::{exec, msg};

unsafe extern "C" fn handle() {
    // ...
    msg::reply_push(b"Part 1").unwrap();
    // ...
    msg::reply_push(b"Part 2").unwrap();
    // ...
    msg::reply_commit_with_gas(42, 0).unwrap();
}
